from IPython.display import display, Markdown
df = pd.read_csv("/Users/claraferrerrodriguez/UOC/Visualitzacio de dades/comtrade_data.csv")
df = df[['reporterISO', 'partnerISO', 'cifvalue']]
df = df.rename(columns={'reporterISO': 'source', 'partnerISO': 'target', 'cifvalue': 'value'})

df['value_mil'] = df['value'] / 1e6


countries = list(pd.unique(df[['source','target']].values.ravel()))

matrix = np.zeros((len(countries), len(countries)))
for _, row in df.iterrows():
    i = countries.index(row['source'])
    j = countries.index(row['target'])
    matrix[i,j] = row['value_mil']

display(Markdown("## Exports 2024"))
fig = ocd.Chord(matrix.tolist(), labels=countries)

fig.show()

france_value = df_1.loc[df_1["partnerISO"] == "France", "cifvalue"].values[0]
rest_value = world_total - france_value

waffle_df = pd.DataFrame({
    "partner": ["France", "Rest of World"],
    "value": [france_value, rest_value]
})


waffle_df["percentage"] = waffle_df["value"] / waffle_df["value"].sum() * 100


total_squares = 100
squares_per_category = np.round(waffle_df["percentage"]).astype(int)

waffle = []
for i, count in enumerate(squares_per_category):
    waffle.extend([waffle_df["partner"][i]] * count)

#matriu
waffle_matrix = np.array(waffle).reshape((10,10))


fig, ax = plt.subplots(figsize=(6,6))
colors = plt.cm.Set3(np.linspace(0,1,len(waffle_df)))

for i in range(10):
    for j in range(10):
        idx = list(waffle_df["partner"]).index(waffle_matrix[i,j])
        ax.add_patch(
            plt.Rectangle((j, 9-i), 1, 1, facecolor=colors[idx], edgecolor='white'))

ax.set_xticks([])
ax.set_yticks([])
ax.set_xlim(0,10)
ax.set_ylim(0,10)
ax.set_title("Spain 2024 Exports: France vs Rest of World")


ax.legend(
    handles=[plt.Rectangle((0,0),1,1,color=colors[i]) for i in range(len(waffle_df))],
    labels=waffle_df["partner"].tolist(),  #  convertir Series a list
    loc="lower center",
    bbox_to_anchor=(0.5,-0.1),
